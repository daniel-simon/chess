let getMoveSquares = require('./AvailableSquaresJudge.js')
let gameConstants = require('./GameConstants.js')
let squareMethods = require('./SquareMethods.js')
 
module.exports = {
 
  getCoordsOfAlliedPieces(color, boardState) {
    let listOfCoords = []
    boardState.forEach((colArray, col) => {
      colArray.forEach((occupant, row) => {
        if (occupant && occupant.color == color) {
          listOfCoords.push([col, row]);
        }
      })
    })
    return (listOfCoords);
  },
 
  getMyUnsafeSquares(myColor, boardState) {
    let unsafeSquares = [];
    let enemyColor = gameConstants.enemyOf(myColor);
    let enemyPieceCoords = this.getCoordsOfAlliedPieces(enemyColor, boardState);
 
    enemyPieceCoords.forEach(enemySquare => {
      let col = enemySquare[0];
      let row = enemySquare[1];
      let occupant = boardState[col][row];
      let squaresThreatenedByThisPiece = getMoveSquares.threatenedByPieceAt(col, row, boardState);
      squaresThreatenedByThisPiece.forEach(square => {
        unsafeSquares.push(square)
      })
    })
    let unsafeSquaresUnique = squareMethods.toUnique(unsafeSquares)
    return (unsafeSquaresUnique)
  }
}
