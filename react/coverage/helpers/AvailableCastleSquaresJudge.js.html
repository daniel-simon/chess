let gameConstants = require('./GameConstants.js')
let attackedSquares = require('./AttackedSquaresJudge.js')
let squareMethods = require('./SquareMethods.js')
 
let castleInfo = {
 
  kingHasNotMoved(moveHistory, player) {
    let kingHasNotMoved = true
    moveHistory.forEach(move => {
      if (move.player == player && move.movedPiece.type == 'king') {
        kingHasNotMoved = false
      }
    })
    return (kingHasNotMoved)
  },
 
  rookHasNotMoved(moveHistory, player, col) {
    let rookHasNotMoved = true
    moveHistory.forEach(move => {
      let fromCol = move.origin[0]
      if (move.player == player) {
        if (move.movedPiece.type == 'rook' && fromCol == col) {
          rookHasNotMoved = false
        } else if (
          move.castlingRook &&
          move.castlingRookOrigin[0] == col
        ) {
          rookHasNotMoved = false
        }
      }
    })
    return (rookHasNotMoved)
  },
 
  getCastlePath(startCol, endCol, player) {
    let row = gameConstants.firstRowFor(player)
    let pathSquares = []
    let colStep = Math.sign(endCol - startCol)
    for (let col = startCol; col != endCol + colStep; col += colStep) {
      pathSquares.push([col, row])
    }
    return (pathSquares)
  },
 
  castlePathIsClear(startCol, endCol, player, boardState) {
    let pathSquares = this.getCastlePath(startCol, endCol, player)
    let clear = true
    let col, row
    pathSquares.forEach(pathSquare => {
      col = pathSquare[0]
      row = pathSquare[1]
      if (boardState[col][row] != null) {
        clear = false
      }
    })
    return (clear)
  },
 
  castlePathIsSafe(startCol, endCol, player, boardState) {
    let pathSquares = this.getCastlePath(startCol, endCol, player)
    let unsafeSquares = attackedSquares.getMyUnsafeSquares(player, boardState)
    let safe = true
    pathSquares.forEach(pathSquare => {
      if (squareMethods.includesSquare(unsafeSquares, pathSquare)) {
        safe = false
      }
    })
    return (safe)
  },
 
  getAvailableCastlingRooks(moveHistory, player, boardState) {
    let rookSquares = []
    let kingSideSquaresToTest = []
    let queenSideSquaresToTest = []
    let kingSide = false
    let queenSide = false
    let homeRow = gameConstants.firstRowFor(player)
    if (
      this.rookHasNotMoved(moveHistory, player, 0) &&
      boardState[0][homeRow] &&
      boardState[0][homeRow].type == 'rook' &&
      boardState[0][homeRow].color == player &&
      this.castlePathIsClear(1, 3, player, boardState) &&
      this.castlePathIsSafe(2, 4, player, boardState)
    ) {
      rookSquares.push([0, homeRow])
    }
    if (
      this.rookHasNotMoved(moveHistory, player, 7) &&
      boardState[7][homeRow] &&
      boardState[7][homeRow].type == 'rook' &&
      boardState[7][homeRow].color == player &&
      this.castlePathIsClear(5, 6, player, boardState) &&
      this.castlePathIsSafe(4, 6, player, boardState)
    ) {
      rookSquares.push([7, homeRow])
    }
    return (rookSquares)
  }
}
 
module.exports = castleInfo
